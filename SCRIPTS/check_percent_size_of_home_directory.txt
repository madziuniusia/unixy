#!/bin/bash

if [ $# -eq 0 ]; then

	echo "Wprowadź procent:"

	read PERCENT

else

	PERCENT=$1

fi



while [ 1 -gt 0 ]; do

	HOMESIZE=`df -T /home | tail -1 | awk '{print $6}'`

	HOMESIZE=`echo $HOMESIZE | rev | cut -c 2- | rev`

	if [ $HOMESIZE -gt $PERCENT ]; then

		echo "Katalog /home zajmuje za dużo miejsca!"

	fi

done
